8. Menu registration (PHP reference)

Three top-level menus + Site Essentials sub-pages.

add_action( 'admin_menu', function () {
  // 1. Site Essentials (top-level)
  add_menu_page(
    'Site Essentials',
    'Site Essentials',
    'manage_options',
    'site-essentials',
    'scos_render_modules_page',
    'dashicons-admin-generic',
    58
  );
  add_submenu_page( 'site-essentials', 'Modules',         'Modules',         'manage_options', 'site-essentials',                'scos_render_modules_page' );
  add_submenu_page( 'site-essentials', 'Agency',          'Agency',          'manage_options', 'site-essentials-agency',         'scos_render_agency_page' );
  add_submenu_page( 'site-essentials', 'Import / Export', 'Import / Export', 'manage_options', 'site-essentials-import-export',  'scos_render_import_export_page' );
  add_submenu_page( 'site-essentials', 'API Keys',        'API Keys',        'manage_options', 'site-essentials-api-keys',       'scos_render_api_keys_page' );
  add_submenu_page( 'site-essentials', 'Cache',           'Cache',           'manage_options', 'site-essentials-cache',          'scos_render_cache_page' );
  add_submenu_page( 'site-essentials', 'Debug',           'Debug',           'manage_options', 'site-essentials-debug',          'scos_render_debug_page' );

  // 2. Content Architecture (top-level — only when Pro module is active)
  if ( scos_module_is_active( 'content_architecture' ) ) {
    add_menu_page(
      'Content Architecture',
      'Content Architecture',
      'manage_options',
      'site-essentials-content-architecture',
      'scos_render_content_architecture_page',
      'dashicons-category',
      59
    );
  }

  // 3. Support (top-level — login redirect target)
  add_menu_page(
    'Support',
    'Support',
    'read',
    'site-essentials-support',
    'scos_render_support_page',
    'dashicons-sos',
    60
  );
} );

// Login redirect to Support
add_filter( 'login_redirect', function ( $redirect, $request, $user ) {
  if ( ! $user instanceof WP_User || empty( $user->roles ) ) return $redirect;
  $option = in_array( 'administrator', $user->roles, true )
    ? get_option( 'se_agency_login_redirect_admin' )
    : get_option( 'se_agency_login_redirect_editor' );
  return $option ? admin_url( $option ) : $redirect;
}, 10, 3 );
